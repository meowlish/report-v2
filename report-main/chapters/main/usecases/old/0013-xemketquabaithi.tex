\textbf{ID} & UC-0013 \\
\hline
\textbf{Tác nhân} & Người dùng đã xác thực \\
\hline
\textbf{Mô tả} & Người dùng xem lại chi tiết kết quả của lượt làm bài đã hoàn thành \\
\hline
\textbf{Điều kiện tiên quyết} & Người dùng đã đăng nhập, lượt làm bài đã được nộp \\
\hline
\textbf{Điều kiện sau} & Kết quả chi tiết được hiển thị \\
\hline
\textbf{Kích hoạt} & Người dùng hoàn thành bài thi hoặc chọn xem lại lượt làm bài cũ \\
\hline
\textbf{Luồng chính} &
\begin{enumerate}
    \item Người dùng chọn xem lại kết quả của một lượt làm bài.
    \item Hệ thống truy vấn dữ liệu đánh giá lượt làm bài.
    \item Hệ thống hiển thị: kết quả từng câu hỏi (đúng/sai), đáp án của người dùng so với đáp án đúng, bảng điểm chi tiết, cấu trúc phần thi.
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item Không có luồng thay thế.
\end{enumerate} \\
\hline
\textbf{Luồng ngoại lệ} &
\begin{enumerate}
    \item \textbf{Lượt làm bài không tồn tại:}
    \begin{enumerate}
        \item[2Err.] Hệ thống trả về lỗi 404 khi lượt làm bài không tồn tại.
    \end{enumerate}
    \item \textbf{Lượt làm bài chưa nộp:}
    \begin{enumerate}
        \item[2Err2.] Hệ thống trả về lỗi khi cố xem kết quả của bài làm chưa nộp.
    \end{enumerate}
\end{enumerate} \\
