\textbf{Luồng chính} &
\begin{enumerate}
    \item Người dùng nhập nội dung tin nhắn.
    \item Người dùng gửi sự kiện 'chat' với \{roomId, message\}.
    \item Hệ thống xác minh người dùng đã tham gia phòng.
    \item Hệ thống phát sự kiện 'message' đến tất cả thành viên khác trong phòng với \{message, uid\}.
    \item Tin nhắn được hiển thị thời gian thực cho tất cả thành viên.
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item Không có luồng thay thế.
\end{enumerate} \\
\hline
\textbf{Luồng ngoại lệ} &
\begin{enumerate}
    \item \textbf{Chưa tham gia phòng:}
    \begin{enumerate}
        \item[3Err.] Hệ thống trả về ForbiddenException vì người dùng chưa tham gia phòng (phải gửi 'join-room' trước).
    \end{enumerate}
    \item \textbf{Mất kết nối WebSocket:}
    \begin{enumerate}
        \item[2Err.] Kết nối WebSocket bị ngắt, tin nhắn không được gửi.
    \end{enumerate}
\end{enumerate} \\
