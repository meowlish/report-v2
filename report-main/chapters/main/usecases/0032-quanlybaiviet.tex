\textbf{Tên Use case} & Quản lý bài viết \\
\hline
\textbf{ID} & UC-0032 \\
\hline
\textbf{Tác nhân} & Người dùng đã xác thực (tác giả), Người dùng bất kỳ (đọc), Quản trị viên (xóa) \\
\hline
\textbf{Mô tả} & Người dùng tạo, chỉnh sửa, xóa bài viết blog; người dùng bất kỳ có thể duyệt và đọc bài viết công khai \\
\hline
\textbf{Điều kiện tiên quyết} & Đăng nhập để tạo/sửa/xóa; không yêu cầu đăng nhập để đọc \\
\hline
\textbf{Điều kiện sau} & Bài viết được tạo, cập nhật, xóa hoặc hiển thị thành công \\
\hline
\textbf{Kích hoạt} & Người dùng mở trang blog \\
\hline
\textbf{Luồng chính} &
\begin{enumerate}
    \item Người dùng mở trang blog.
    \item Hệ thống hiển thị danh sách bài viết phân trang với: tiêu đề, tác giả, thẻ phân loại, ngày tạo.
    \item Người dùng nhấn Tạo bài viết mới.
    \item Người dùng nhập: tiêu đề (bắt buộc), nội dung (bắt buộc), thẻ phân loại.
    \item Người dùng nhấn Xuất bản.
    \item Hệ thống tạo bài viết với người dùng là tác giả.
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item \textbf{Chỉnh sửa bài viết:}
    \begin{enumerate}
        \item[3A.] Tác giả mở bài viết, nhấn Chỉnh sửa, thay đổi tiêu đề/nội dung/thẻ phân loại, nhấn Lưu.
    \end{enumerate}
    \item \textbf{Xóa bài viết:}
    \begin{enumerate}
        \item[3B.] Tác giả hoặc Quản trị viên nhấn Xóa, xác nhận. Hệ thống xóa bài viết.
    \end{enumerate}
    \item \textbf{Đọc bài viết:}
    \begin{enumerate}
        \item[2C.] Người dùng bất kỳ chọn bài viết để xem nội dung đầy đủ, lọc theo tác giả hoặc thẻ phân loại.
    \end{enumerate}
\end{enumerate} \\
\hline
\textbf{Luồng ngoại lệ} &
\begin{enumerate}
    \item \textbf{Bài viết không tồn tại:}
    \begin{enumerate}
        \item[Err1.] Hệ thống trả về lỗi 404.
    \end{enumerate}
    \item \textbf{Không có quyền:}
    \begin{enumerate}
        \item[Err2.] Hệ thống trả về lỗi 403 khi người dùng không phải tác giả hoặc Admin.
    \end{enumerate}
\end{enumerate} \\
\hline
