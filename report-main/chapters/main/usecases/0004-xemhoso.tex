        \item[2A.] Hệ thống truy vấn thông tin công khai của người dùng đó.
        \item[3A.] Hệ thống hiển thị hồ sơ giới hạn: tên người dùng, họ tên, tiểu sử, ảnh đại diện.
    \end{enumerate}
