\textbf{Điều kiện tiên quyết} & Người dùng đã đăng nhập và có ít nhất hai phương thức xác thực đã liên kết \\
\hline
\textbf{Điều kiện sau} & Phương thức xác thực được xóa khỏi tài khoản \\

\textbf{Kích hoạt} & Người dùng chọn xóa một phương thức xác thực \\
\hline
\textbf{Luồng chính} &
\begin{enumerate}
    \item Người dùng mở trang cài đặt tài khoản.
    \item Hệ thống hiển thị danh sách các phương thức xác thực đã liên kết.
    \item Người dùng chọn phương thức cần xóa và nhấn Xóa.
    \item Hệ thống yêu cầu xác nhận.
    \item Người dùng xác nhận xóa.
    \item Hệ thống xóa phương thức xác thực và xác nhận thành công.
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item Không có luồng thay thế.
\end{enumerate} \\
\hline
\textbf{Luồng ngoại lệ} &
\begin{enumerate}
    \item \textbf{Không thể xóa phương thức cuối cùng:}
    \begin{enumerate}
        \item[6Err.] Hệ thống trả về lỗi thông báo không thể xóa phương thức xác thực cuối cùng vì sẽ khiến tài khoản bị khóa.
    \end{enumerate}
    \item \textbf{Phương thức không tồn tại:}
    \begin{enumerate}
        \item[3Err.] Hệ thống trả về lỗi 404 thông báo phương thức xác thực không tồn tại.
    \end{enumerate}
\end{enumerate} \\
