\textbf{Tên Use case} & Đặt lịch gọi video qua Google Meet \\
\hline
\textbf{ID} & UC-0036 \\
\hline
\textbf{Tác nhân} & Học viên đã xác thực \\
\hline
\textbf{Mô tả} & Học viên đặt lịch gọi video với giáo viên, hệ thống tạo cuộc họp Google Meet \\
\hline
\textbf{Điều kiện tiên quyết} & Học viên đã đăng nhập; cả học sinh và mentor đều phải connect tài khoản với Google Calendar; thời gian chọn giờ họp không trùng với lịch bận của mentor \\
\hline
\textbf{Điều kiện sau} & Lịch hẹn được tạo; trạng thái xác nhận; liên kết Google Meet được lưu \\
\hline
\textbf{Kích hoạt} & Học viên chọn Đặt lịch gọi video trong ứng dụng \\
\hline
\textbf{Luồng chính} &
\begin{enumerate}
    \item Học viên mở màn hình đặt lịch và chọn giáo viên.
    \item Hệ thống hiển thị lịch rảnh theo ngày và khung giờ.
    \item Học viên chọn ngày và khung giờ phù hợp.
    \item Hệ thống kiểm tra tính hợp lệ của giáo viên và khung giờ.
    \item Hệ thống tạo lịch hẹn và gửi yêu cầu tạo cuộc họp Google Meet.
    \item Hệ thống lưu thông tin cuộc họp và xác nhận lịch hẹn.
    \item Hệ thống hiển thị thông tin lịch hẹn và link Google Meet cho học viên.
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item \textbf{Không có lịch rảnh:}
    \begin{enumerate}
        \item[2A.] Hệ thống hiển thị thông báo không có khung giờ trống cho ngày đã chọn.
        \item[3A.] Học viên chọn ngày khác hoặc giáo viên khác.
    \end{enumerate}
    \item \textbf{Đặt lịch ngoài giờ hợp lệ:}
    \begin{enumerate}
        \item[3B.] Học viên chọn khung giờ không hợp lệ.
        \item[4B.] Hệ thống yêu cầu chọn lại khung giờ hợp lệ.
    \end{enumerate}
\end{enumerate} \\
\hline
\textbf{Luồng ngoại lệ} &
\begin{enumerate}
    \item \textbf{Giáo viên không hỗ trợ video call:}
    \begin{enumerate}
        \item[4Err.] Hệ thống trả về lỗi giáo viên không khả dụng cho video call.
    \end{enumerate}
    \item \textbf{Khung giờ đã được đặt:}
    \begin{enumerate}
        \item[4Err.] Hệ thống trả về lỗi khung giờ đã có lịch hẹn.
    \end{enumerate}
    \item \textbf{Không thể tạo Google Meet:}
    \begin{enumerate}
        \item[5Err.] Hệ thống báo lỗi tạo cuộc họp thất bại và đề nghị thử lại.
    \end{enumerate}
\end{enumerate} \\
