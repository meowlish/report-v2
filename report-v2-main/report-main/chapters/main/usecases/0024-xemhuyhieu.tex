\textbf{Luồng chính} &
\begin{enumerate}
    \item Người dùng mở trang huy hiệu.
    \item Hệ thống hiển thị danh sách tất cả huy hiệu có sẵn với tên, tên hiển thị và mô tả.
\end{enumerate} \\

\textbf{Luồng thay thế} &
\begin{enumerate}
    \item \textbf{Xem huy hiệu cá nhân:}
    \begin{enumerate}
        \item[1A.] Người dùng đã xác thực xem huy hiệu đã đạt được của mình (phân trang cursor-based).
    \end{enumerate}
    \item \textbf{Xem huy hiệu người dùng khác:}
    \begin{enumerate}
        \item[1B.] Người dùng xem huy hiệu đã đạt được của người dùng khác theo ID.
    \end{enumerate}
    \item \textbf{Xem tiến trình huy hiệu:}
    \begin{enumerate}
        \item[1C.] Người dùng đã xác thực xem tiến trình: chuỗi đăng nhập (hiện tại/dài nhất/tổng), thống kê nộp bài (điểm tốt/điểm hoàn hảo/tổng).
    \end{enumerate}
\end{enumerate} \\

\begin{enumerate}
    \item \textbf{Người dùng không tồn tại:}
    \begin{enumerate}
        \item[1BErr.] Hệ thống trả về lỗi 404 khi xem huy hiệu của người dùng không tồn tại.
    \end{enumerate}
