\textbf{Kích hoạt} & Quản trị viên chọn xử lý báo cáo \\
\hline
\textbf{Luồng chính} &
\begin{enumerate}
    \item Quản trị viên mở trang quản lý báo cáo.
    \item Quản trị viên chọn báo cáo ở trạng thái pending hoặc reviewing.
    \item Quản trị viên xem xét nội dung được báo cáo.
    \item Quản trị viên thay đổi trạng thái (pending → reviewing → resolved/rejected).
    \item Quản trị viên viết phản hồi (adminResponse) giải thích hành động đã thực hiện.
    \item Quản trị viên nhấn Cập nhật.
    \item Hệ thống cập nhật báo cáo, ghi nhận thời gian cập nhật.
    \item Hệ thống tạo thông báo cho người gửi báo cáo (UC-0046).
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item \textbf{Người dùng cập nhật báo cáo của mình:}

        \item[2A.] Người dùng chỉ có thể chỉnh sửa tiêu đề/mô tả của báo cáo khi trạng thái còn "pending".
        \item[3A.] Người dùng nhấn Lưu, hệ thống cập nhật báo cáo.
    \end{enumerate}
\end{enumerate} \\
