\textbf{Tên Use case} & Tham gia phòng chat \\
\hline
\textbf{ID} & UC-0026 \\
\hline
\textbf{Tác nhân} & Người dùng đã xác thực \\
\hline
\textbf{Mô tả} & Người dùng tham gia một phòng chat để nhận và gửi tin nhắn thời gian thực qua WebSocket \\
\hline
\textbf{Điều kiện tiên quyết} & Người dùng đã đăng nhập, phòng chat tồn tại \\
\hline
\textbf{Điều kiện sau} & Người dùng đã tham gia phòng và có thể nhận/gửi tin nhắn \\
\hline
\textbf{Kích hoạt} & Người dùng chọn tham gia một phòng chat \\
\hline
\textbf{Luồng chính} &
\begin{enumerate}
    \item Người dùng mở ứng dụng chat.
    \item Client kết nối WebSocket đến endpoint (/chat/ws/socket.io).
    \item Gateway xác thực JWT, trích xuất uid từ token và chuyển tiếp đến dịch vụ chat.
    \item Kết nối được thiết lập, hệ thống tự động thêm người dùng vào phòng riêng tư (uid).
    \item Người dùng gửi sự kiện 'join-room' với roomId.
    \item Hệ thống thêm người dùng vào phòng chỉ định.
    \item Người dùng bắt đầu nhận tin nhắn từ phòng.
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item \textbf{Rời phòng chat:}
    \begin{enumerate}
        \item[7A.] Người dùng gửi sự kiện 'leave-room' với roomId.
        \item[8A.] Hệ thống xóa người dùng khỏi phòng, người dùng không nhận tin nhắn từ phòng nữa.
    \end{enumerate}
\end{enumerate} \\
\hline
\textbf{Luồng ngoại lệ} &
\begin{enumerate}
    \item \textbf{Kết nối WebSocket thất bại:}
    \begin{enumerate}
        \item[3Err.] Không có header xác thực, hệ thống ngắt kết nối.
    \end{enumerate}
    \item \textbf{Phòng không tồn tại:}
    \begin{enumerate}
        \item[5Err.] Hệ thống trả về lỗi khi phòng không tồn tại.
    \end{enumerate}
    \item \textbf{Người dùng bị cấm:}
    \begin{enumerate}
        \item[5Err2.] Hệ thống trả về lỗi 403 khi người dùng bị cấm khỏi phòng.
    \end{enumerate}
\end{enumerate} \\
\hline
