\textbf{Tên Use case} & Quản lý người dùng \\
\hline
\textbf{ID} & UC-0023 \\
\hline
\textbf{Tác nhân} & Quản trị viên (Admin) \\
\hline
\textbf{Mô tả} & Quản trị viên quản lý tài khoản người dùng, gán và thu hồi vai trò \\
\hline
\textbf{Điều kiện tiên quyết} & Người dùng đã đăng nhập và có vai trò Admin \\
\hline
\textbf{Điều kiện sau} & Vai trò người dùng được gán hoặc thu hồi thành công \\
\hline
\textbf{Kích hoạt} & Quản trị viên mở trang quản lý người dùng \\
\hline
\textbf{Luồng chính} &
\begin{enumerate}
    \item Quản trị viên mở trang quản lý người dùng.
    \item Hệ thống hiển thị danh sách người dùng phân trang (lọc theo tên, ID, vai trò, quyền hạn).
    \item Quản trị viên chọn một người dùng.
    \item Hệ thống hiển thị thông tin đầy đủ của người dùng.
\end{enumerate} \\
\hline
\textbf{Luồng thay thế} &
\begin{enumerate}
    \item \textbf{Gán vai trò:}
    \begin{enumerate}
        \item[4A.] Quản trị viên chọn gán vai trò cho người dùng.
        \item[5A.] Quản trị viên chọn vai trò cần gán (từ danh sách vai trò có sẵn).
        \item[6A.] Hệ thống gán vai trò và xác nhận thành công.
    \end{enumerate}
    \item \textbf{Thu hồi vai trò:}
    \begin{enumerate}
        \item[4B.] Quản trị viên chọn thu hồi vai trò từ người dùng.
        \item[5B.] Hệ thống xóa vai trò và xác nhận.
    \end{enumerate}
    \item \textbf{Xem danh sách vai trò và quyền:}
    \begin{enumerate}
        \item[1C.] Quản trị viên xem danh sách tất cả vai trò có sẵn cùng tập quyền hạn.
        \item[1D.] Quản trị viên xem danh sách tất cả quyền hạn có sẵn.
    \end{enumerate}
\end{enumerate} \\
\hline
\textbf{Luồng ngoại lệ} &
\begin{enumerate}
    \item \textbf{Người dùng không tồn tại:}
    \begin{enumerate}
        \item[3Err.] Hệ thống trả về lỗi 404.
    \end{enumerate}
    \item \textbf{Không có quyền:}
    \begin{enumerate}
        \item[1Err.] Hệ thống trả về lỗi 403 khi người dùng không có vai trò Admin.
    \end{enumerate}
    \item \textbf{Vai trò không tồn tại:}
    \begin{enumerate}
        \item[5AErr.] Hệ thống trả về lỗi khi vai trò không tồn tại.
    \end{enumerate}
\end{enumerate} \\
